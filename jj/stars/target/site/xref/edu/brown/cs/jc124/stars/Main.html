14  /**
15   * @author jchen

17   *         The Main class for searching by stars.
18   */
19  public final class Main {
20    /**
21     * Executes the stars querier based on command line arguments.
22     *
23     * @param args
24     *          the passed in arguments, in the form 'program [--gui]
25     *          /path/to/csv'
26     */
27    public static void main(String[] args) {
28      OptionParser parser = new OptionParser();
29  
30      parser.accepts("gui");
31      OptionSpec<File> fileSpec = parser.nonOptions().ofType(File.class);
32      OptionSet options = parser.parse(args);
33  
34      File in = options.valueOf(fileSpec);
35      if (in == null) {
36        System.out.println("ERROR: Please specify a star CSV file");
37        System.exit(1);
38      }
39  
40      StarDataManager stars = init(in);
41      if (options.has("gui")) {
42        gui(stars, in);
43      } else {
44        cli(stars);
45      }
46    }
47  
48    private static StarDataManager init(File csv) {
49      List<Star> csvData = new ArrayList<Star>();
50      CsvReader cr = null;
51      boolean notEof = false;
52      try {
53        cr = new CsvReader(csv);
54        cr.setHeader(new String[] {"StarID", "ProperName", "X", "Y", "Z"});
55        notEof = cr.readLine();
56      } catch (IOException e) {
57        System.out.println("ERROR: " + e.getMessage());
58        System.exit(1);
59      }
60  
61      while (notEof) {
62        int id;
63        String name;
64        double x, y, z;
65  
66        try {
67          id = Integer.parseInt(cr.get("StarID"));
68          name = cr.get("ProperName");
69          x = Double.parseDouble(cr.get("X"));
70          y = Double.parseDouble(cr.get("Y"));
71          z = Double.parseDouble(cr.get("Z"));
72        } catch (NumberFormatException e) {
73          System.out.println("ERROR: invalid CSV data parsed");
74          continue;
75        }
76  
77        csvData.add(new Star(id, name, x, y, z));
78  
79        try {
80          notEof = cr.readLine();
81        } catch (IOException e) {
82          System.out.println("ERROR: " + e.getMessage());
83          notEof = true;
84        }
85      }
86  
87      try {
88        cr.close();
89      } catch (IOException e) {
90        System.out.println("ERROR: " + e.getMessage());
91        System.exit(1);
92      }
93  
94      return new StarDataManager(csvData);
95    }
96  
97    private static void gui(StarDataManager stars, File db) {
98      new SparkServer(stars, db).runSparkServer();
99    }
100 
101   private static void cli(StarDataManager stars) {
102     new CommandLine(stars).runCli();
103   }
104 
105   private Main() {
106   }
