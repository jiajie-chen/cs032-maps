3   import java.io.BufferedReader;
4   import java.io.IOException;
5   import java.io.InputStreamReader;
6   import java.util.List;
7   import java.util.NoSuchElementException;
8   
9   /**
10   * @author jchen
11   *
12   *         A class to manage and execute command line queries on a Star KdTree.
13   */
14  public class CommandLine {
15    private StarDataManager stars;
16  
17    /**
18     * Creates a command line executer that queries the given database.
19     *
20     * @param stars
21     *          the storage of stars to use for querying
22     */
23    public CommandLine(StarDataManager stars) {
24      this.stars = stars;
25    }
26  
27    /**
28     * runs a REPL that queries the DB and outputs to the console.
29     */
30    public void runCli() {
31      BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
32  
33      String in = null;
34      try {
35        in = br.readLine();
36      } catch (IOException e) {
37        System.out.println("ERROR: can't read input stream");
38      }
39      while (in != null) {
40        List<Star> results = cliCommand(in);
41  
42        if (results != null) {
43          for (Star s : results) {
44            System.out.println(s.getId());
45          }
46        }
47  
48        try {
49          in = br.readLine();
50        } catch (IOException e) {
51          System.out.println("ERROR: can't read input stream");
52        }
53      }
54    }
55  
56    private List<Star> cliCommand(String in) {
57      // parse out parameters, convert to data types
58      String[] params = in.trim().split("\\s+");
59      // quote handling
60      if (params.length >= 3 && params[2].charAt(0) == '"') {
61        String quote = "";
62        for (int i = 2; i < params.length; i++) {
63          quote += params[i];
64        }
65        params = new String[] {params[0], params[1], quote};
66      }
67  
68      // length check, then execute command, and capture any parsing errors
69      try {
70        if (params.length == 5) {
71          return stars.locationCommand(params[0], params[1], params[2],
72              params[3], params[4]);
73        } else if (params.length == 3) {
74          return stars.nameCommand(params[0], params[1], params[2]);
75        } else {
76          System.out.println("ERROR: incorrect number of parameters");
77          return null;
78        }
79      } catch (NoSuchElementException | IllegalArgumentException
80          | IndexOutOfBoundsException e) {
81        // only errors caused by parsing or searching tree
82        System.out.println("ERROR: " + e.getMessage());
83        return null;
84      }
85    }
86  }
