1   package edu.brown.cs.jc124.stars;
2   
3   import edu.brown.cs.jc124.kdtree.CartesianCoordinate;
4   
5   /**
6    * @author jchen
7    *
8    *         An extension of CartesianCoordinate for additionally holding
9    *         star-specific data.
10   */
11  public class Star extends CartesianCoordinate {
12    private int starId;
13    private String name;
14  
15    /**
16     * Makes a new star with a name and id at the given x, y, z coordinates.
17     *
18     * @param starId
19     *          the star's id
20     * @param name
21     *          the name of the star
22     * @param x
23     *          the x coordinate of its location
24     * @param y
25     *          the y coordinate of its location
26     * @param z
27     *          the z coordinate of its location
28     */
29    public Star(int starId, String name, double x, double y, double z) {
30      super(3, x, y, z);
31      this.starId = starId;
32      this.name = name;
33    }
34  
35    /**
36     * Gets the star's id.
37     *
38     * @return the star's id
39     */
40    public int getId() {
41      return starId;
42    }
43  
44    /**
45     * Gets the star's name.
46     *
47     * @return the star's name
48     */
49    public String getName() {
50      return name;
51    }
52  
53    @Override
54    public String toString() {
55      return "[" + name + ", " + starId + ", <" + getField(0) + ", "
56          + getField(1) + ", " + getField(2) + ">]";
57    }
58  }
