39    /**
40     * Makes a new server that contains the stars in the file db.
41     *
42     * @param stars
43     *          the stars to query from
44     * @param db
45     *          the file the DB was generated from
46     */
47    public SparkServer(StarDataManager stars, File db) {
48      this.stars = stars;
49      this.db = db;
50    }
51  
52    /**
53     * runs the GUI server on port 2345.
54     */
55    public void runSparkServer() {
56      Spark.setPort(2345);
57  
58      Spark.externalStaticFileLocation("src/main/resources/static");
59      Spark.exception(Exception.class, new ExceptionPrinter());
60  
61      FreeMarkerEngine freeMarker = createEngine();
62  
63      // Setup Spark Routes
64  
65      // front end
66      Spark.get("/", new FrontHandler(), freeMarker);
67  
68      // RESTful API
69      Spark.get("/name/:command/:kr/:starname", new NameResultsHandler());
70      Spark.get("/coord/:command/:kr/:x/:y/:z", new LocationResultsHandler());
71    }
72  
73    private static FreeMarkerEngine createEngine() {
74      Configuration config = new Configuration();
75      File templates = new File("src/main/resources/spark/template/freemarker");
76      try {
77        config.setDirectoryForTemplateLoading(templates);
78      } catch (IOException ioe) {
79        System.out.printf("ERROR: Unable use %s for template loading.\n",
80            templates);
81        System.exit(1);
82      }
83      return new FreeMarkerEngine(config);
84    }

99    /**
100    * @author jchen
101    * Helper class for storing and formulating json responses.
102    */
103   private static final class StarJsonResponse {
104     // these are only for serialization, so keep unused
105     @SuppressWarnings("unused")
106     private int id;
107     @SuppressWarnings("unused")
108     private String name;
109 
110     private StarJsonResponse(int id, String name) {
111       this.id = id;
112       this.name = name;
113     }
114 
115     private static String toStarJson(List<Star> results) {
116       List<StarJsonResponse> ids = new ArrayList<StarJsonResponse>();
117 
118       for (Star s : results) {
119         ids.add(new StarJsonResponse(s.getId(), s.getName()));
120       }
121 
122       Type listType = new TypeToken<List<StarJsonResponse>>() {}.getType();
123 
124       JsonObject res = new JsonObject();
125       res.addProperty("success", true);
126       res.add("stars", GSON.toJsonTree(ids, listType));
127       return GSON.toJson(res);
128     }
129 
130     private static String toErrorJson(String msg) {
131       JsonObject err = new JsonObject();
132       err.addProperty("success", false);
133       err.addProperty("error", msg);
134       return GSON.toJson(err);
135     }
136   }

138   /**
139    * @author jchen
140    * Handles name based queries.
141    */
142   private class NameResultsHandler implements Route {
143     @Override
144     public Object handle(final Request req, final Response res) {
145       String command = req.params(":command");
146       String kr = req.params(":kr");
147       String name = req.params(":starname");
148 
149       // wrap name in quotes to adapt to nameCommand name parameter
150       name = "\"" + name + "\"";
151 
152       // try to create response, return error message if exception
153       try {
154         List<Star> results = stars.nameCommand(command, kr, name);
155         return StarJsonResponse.toStarJson(results);
156       } catch (NoSuchElementException | IllegalArgumentException
157           | IndexOutOfBoundsException e) {
158         // only errors caused by parsing or searching tree
159         return StarJsonResponse.toErrorJson(e.getMessage());
160       }
161     }
162   }

164   /**
165    * @author jchen
166    * Handles location based queries.
167    */
168   private class LocationResultsHandler implements Route {
169     @Override
170     public Object handle(final Request req, final Response res) {
171       String command = req.params(":command");
172       String kr = req.params(":kr");
173       String x = req.params(":x");
174       String y = req.params(":y");
175       String z = req.params(":z");
176 
177       // try to create response, return error message if exception
178       try {
179         List<Star> results = stars.locationCommand(command, kr, x, y, z);
180         return StarJsonResponse.toStarJson(results);
181       } catch (NoSuchElementException | IllegalArgumentException
182           | IndexOutOfBoundsException e) {
183         // only errors caused by parsing or searching tree
184         return StarJsonResponse.toErrorJson(e.getMessage());
185       }
186     }
187   }

189   /**
190    * @author jchen
191    * Handles java exceptions.
192    */
193   private static class ExceptionPrinter implements ExceptionHandler {
194     @Override
195     public void handle(Exception e, Request req, Response res) {
196       res.status(500);
197       StringWriter stacktrace = new StringWriter();
198       try (PrintWriter pw = new PrintWriter(stacktrace)) {
199         pw.println("<pre>");
200         e.printStackTrace(pw);
201         pw.println("</pre>");
202       }
203       res.body(stacktrace.toString());
204     }
